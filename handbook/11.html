    电梯项目--移动端地图电梯打点leaflet.js聚合展示

    2023/7/24 地图开发leaflet.j

    # 问题背景

    在移动端中因为青岛市电梯的风险打点高达 7w 部，使用 leaflet 组件打点的时候会卡顿 3-4 分钟才会打出来，而且地图会很卡，为了解决该问题，与后端沟通最终确定了解决方案，首先返回各区域的电梯总量展示在前端，点击后再进入该区域的街道电梯数量，在点击街道后再调用电梯打点接口，在地图上打点，由于各街道的电梯数量没那么多所以可以解决该问题，而且最终的打点需要有聚合效果，在网上翻了下 leaflet 的相关资料后，发现该组件官方有聚合插件 MarkerCluster，使用即可

    移动端使用 Vue3+Ts

    # 解决问题

    • 首先进入页面后在 mounted 中执行 addMarkersNum 方法，获取青岛市各区域的电梯数量，再讲电梯数量数据添加到青岛市的地理文件中，循环在中心点打点时展示出来电梯数量即可，使用 L.divIcon 方法创建展示的内容框 on('click')添加点击事件，进入街道的打点，如果是青岛市级别则打区域数量点，如果是区域级别就打街道的点
    const addMarkersNum = async () => {
  //获取青岛市各区或各街道数量
  const { code, data } = await request({
    method: 'get',
    url: '获取青岛市各区或各街道数量接口',
    params: {
      riskLevel: params.riskLevel,
      regionCode: params.regionCode == '370200' ? null : params.regionCode,
    },
  })
  layerIndex.forEach(item => {
    item.removeFrom(mapInstance)
  })
  if (code == '00000') {
    list = params.regionCode
      ? data.filter(
        item =>
          item.regionCode != null &&
            item.regionCode.indexOf(regionCode) !== -1,
      )
      : data
  }
  //青岛市级别
  if (params.regionCode == '370200') {
    //把区县数量加入到features中，以便生成地图上的div
    features.forEach(v => {
      if (data.length > 0) {
        data.forEach(item => {
          if (item.regionCode == v.properties.adcode) {
            areaValue.push(item)
            v.properties.riskNum = item.riskNum || 0
          }
        })
      } else {
        v.properties.riskNum = 0
      }
    })
    mapInstance.invalidateSize(true)
    markerLayers = features.map(
      (
        {
          properties: {
            adcode,
            name,
            riskNum,
            center: [lng, lat],
          },
        },
        index,
      ) => {
        layerIndex[index] = L.marker([lat, lng], {
          icon: L.divIcon({
            className: 'lefalet-marker',
            html: `<div class="lefalet-marker-content">${name}共计${riskNum}台</div>`,
          }),
        }).addTo(mapInstance)
        layerIndex[index].reginoCode = adcode
        layerIndex[index].on('click', e => {
          currCode =
          currCode === e.target.reginoCode ? null : e.target.reginoCode
          mapInstance.setView([lat, lng], 12)
          layerIndex.forEach(item => {
            item.removeFrom(mapInstance)
          })
          regionCode.value = e.target.reginoCode
          params.regionCode = e.target.reginoCode
          //点击清空图标并且重新打点
          console.log('点击了')
          resetMarkers(currCode)
        })
      },
    )
  }
  //青岛街道级
  else {
    layerIndex.forEach(item => {
      item.removeFrom(mapInstance)
    })
    const { data } = await request({'地区/街道电梯总数量接口'})
    mapPointStreet(params.regionCode, data)
  }
}
    • 街道打点与区域打点基本一致，只是因为本地没有青岛市各街道的中心位置，所以借助 turf 插件传入各街道地理数据后计算获得，再将 div 添加到街道的中心点上展示数量，如果点击了街道就直接进行电梯地图聚合打点
    //街道打点
const mapPointStreet = (reginSteet, data) => {
  const list = reginSteet
    ? data.filter(
        (item) =>
          item.regionCode != null && item.regionCode.indexOf(reginSteet) !== -1
      )
    : data;
  const streetFeatures = [];

  list.forEach(async (item) => {
    if (item.regionCode) {
      const parentRegionCode =
        item.regionCode.length > 6
          ? item.regionCode.substring(0, 6)
          : item.regionCode;

      if (item.regionCode.length > 6) {
        const street = streets[Number.parseInt(parentRegionCode)].filter(
          ({ properties: { code } }) => code === item.regionCode
        );

        const poly = polygon(street[0].geometry.coordinates[0]); //获得区域
        //使用turf插件获取街道中心点
        street[0].properties.adcode = street[0].properties.code;
        street[0].properties.center = centroid(poly).geometry.coordinates; //区域中心点坐标

        streetFeatures.push(...street);
      }
    }
  });

  //加上各街道风险数量
  streetFeatures.forEach((v) => {
    if (data.length > 0) {
      data.forEach((item) => {
        if (item.regionCode == v.properties.adcode) {
          areaValue.push(item);
          v.properties.riskNum = item.riskNum || 0;
        }
      });
    } else {
      v.properties.riskNum = 0;
    }
  });

  mapInstance.invalidateSize(true);
  markerLayers = streetFeatures.map(
    (
      {
        properties: {
          adcode,
          name,
          riskNum,
          center: [lng, lat],
        },
      },
      index
    ) => {
      layerIndex[index] = L.marker([lat, lng], {
        icon: L.divIcon({
          className: "lefalet-marker",
          html: `<div class="lefalet-marker-content">${name}共计${riskNum}台</div>`,
        }),
      }).addTo(mapInstance);
      layerIndex[index].reginoCode = adcode;
      //街道点位的点击事件
      layerIndex[index].on("click", (e) => {
        //清空点位
        layerIndex.forEach((item) => {
          item.removeFrom(mapInstance);
        });

        fetchRiskList(e.target.reginoCode);
      });
    }
  );
};
    • 在 fetchRiskList 方法里，先 idMap.clear()清空之前的电梯数量展示 Div 打点，然后调用接口获得各电梯的经纬度等数据，使用 L.markerClusterGroup 进行聚合打点，绑定聚合点位点击后自动散开事件 on('clusterclick'),再绑定图标点击后弹出电梯详情弹窗事件即可，用 marker 保存 L.marker 返回的对象，然后保存电梯的 ID 和风险 ID 以便后续详情弹窗调接口使用
    //街道后直接在地图上打电梯的打点
const fetchRiskList = async (code) => {
  idMap.clear();
  showLoadingToast({
    duration: 0,
    forbidClick: true,
    message: "正在获取数据",
  });
  const res = await api.liftOnlineApi.appLiftRiskMonitor.getLiftRiskMap.request(
    {
      riskLevel: params.riskLevel,
      regionCode: code,
    }
  );

  if (cluster) {
    cluster.clearLayers();
  }

  cluster = L.markerClusterGroup({
    showCoverageOnHover: false,
    zoomToBoundsOnClick: false,
    removeOutsideVisibleBounds: true,
    maxZoom: 18,
    maxClusterRadius: 50,
    spiderfyOnMaxZoom: true,
  });
  // 聚合点点击事件，点击展开点
  cluster.on("clusterclick", function (e) {
    e.layer.zoomToBounds();
  });
  // 绑定 Marker Cluster 组的点击事件,点击出弹窗
  cluster.on("click", ({ sourceTarget }) => {
    showCard.value = true;
    fetchRiskDetail(sourceTarget.options.liftId, sourceTarget.options.riskId);
  });
  res.data.forEach(({ latitude, longitude, liftId, riskId }) => {
    const marker = L.marker([latitude, longitude], { icon: DefaultIcon });
    marker.options.liftId = liftId;
    marker.options.riskId = riskId;
    cluster.addLayer(marker);
    //塞入数据即可，回调可以返回
  });
  mapInstance.addLayer(cluster);
  mapInstance.setView(mapInstance.getCenter());
  closeToast();
};
