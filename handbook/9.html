    排水项目--GIS调度页面监控展示

    2023/7/24 地图开发

    # 问题背景

    之前排水详情页中有个 GIS 调度页面，里面可以根据点位选择距离 500m 1km 内的相关设施打点并查看，最近产品突发奇想增加了个附近监控的查看，最开始使用的是 h5 自带的 video 标签播放.u3m8 的监控地址，但由于效果不好，于是换成了 Jessibuca 播放器

    • 监控组件的代码
    
<template>
  <div id="container" ref="container" class="player-panel"></div>
</template>
<script>
import { uuid } from '@/cim-map/utils/uuid.js'
export default {
  props: {
    videoUrl: {
      type: String,
      default: '',
    },
  },
  data() {
    return {
      useWCS: false,
      useMSE: true,
      showOperateBtns: false,
      showBandwidth: false,
      jessibuca: '',
      // videoUrl: '',
      player: '',
      id: uuid(),
      fullId: uuid(),
      bLoading: false,
      isAbazhou: false,
      // 是否是全屏
      isFullSceen: false,
      fullscreen: false,
      lastControlType: null,
      isHistory: false,
    }
  },
  watch: {
    videoUrl() {
      if (this.videoUrl) {
        this.players()
      }
    },
  },
  mounted() {
    this.create()
  },
  //销毁
  beforeDestroy() {
    this.closeVideo()
  },
  methods: {
    create(options) {
      const showOperateBtns = true
      options = options || {}
      this.jessibuca = new window.JessibucaPro(
        Object.assign(
          {
            container: this.$refs.container,

            isResize: false,
            supportDblclickContainerFullscreen: true,
            isFlv: true,
            debug: false,
            useMSE: false,
            useSIMD: true,
            useWCS: false,
            debugLevel: 'debug',
            hasAudio: false,
            showBandwidth: showOperateBtns, // 显示网速
            showPerformance: false, // 显示性能
            operateBtns: {
              fullscreen: showOperateBtns,
              screenshot: showOperateBtns,
              play: showOperateBtns,
              audio: showOperateBtns,
              record: false,
              ptz: false,
              quality: false,
              performance: false,
            },
            watermarkConfig: {
              text: {
                content: 'jessibuca-pro',
              },
              right: 10,
              top: 10,
            },
          },
          options,
        ),
      )

      var _this = this
      this.jessibuca.on('load', function () {
        console.log('on load')
      })
      this.jessibuca.on('log', function (msg) {
        console.log('on log', msg)
      })
      this.jessibuca.on('record', function (msg) {
        console.log('on record:', msg)
      })
      this.jessibuca.on('pause', function () {
        console.log('on pause')

        _this.playing = false
      })
      this.jessibuca.on('play', function () {
        console.log('on play')
        _this.playing = true
      })
      this.jessibuca.on('fullscreen', function (msg) {
        console.log('on fullscreen', msg)
      })
      this.jessibuca.on('mute', function (msg) {
        console.log('on mute', msg)
        _this.quieting = msg
      })
      this.jessibuca.on('mute', function (msg) {
        console.log('on mute2', msg)
      })
      this.jessibuca.on('audioInfo', function (msg) {
        console.log('audioInfo', msg)
      })
      this.jessibuca.on('videoInfo', function (info) {
        console.log('videoInfo', info)
      })
      this.jessibuca.on('error', function (error) {
        _this.$message.error('此监控流无法播放')
        console.log('error', error)
      })
      this.jessibuca.on('timeout', function () {
        // console.log('timeout')
      })
      this.jessibuca.on('start', function () {
        // console.log('frame start')
      })
      this.jessibuca.on('performance', function (performance) {
        var show = '卡顿'
        if (performance === 2) {
          show = '非常流畅'
        } else if (performance === 1) {
          show = '流畅'
        }
        _this.performance = show
      })
      this.jessibuca.on('buffer', function (buffer) {
        // console.log('buffer', buffer)
      })
      this.jessibuca.on('stats', function (stats) {
        // console.log('stats', stats)
      })
      this.jessibuca.on('kBps', function (kBps) {
        // console.log('kBps', kBps)
      })
      this.jessibuca.on('play', () => {
        this.playing = true
        this.loaded = true
        this.quieting = this.jessibuca.isMute()
      })
      this.players()
    },

    players() {
      this.jessibuca

        .play(this.videoUrl)
        .then(() => {})
        .catch(e => {
          this.$message.error('此监控流无法播放')
          console.error(e)
        })
    },

    closeVideo() {
      this.jessibuca?.close()
    },
  },
}
</script>
